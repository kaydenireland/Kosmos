#version 460 core

in vec3 chunkPos;
in vec3 color;
in vec2 textureCoord;

out vec3 passColor;
out vec2 passTextureCoord;

uniform mat4 projection;
uniform mat4 model;
uniform mat4 view;

void main() {
	gl_Position = projection * view * model * vec4(chunkPos, 1.0);
	passColor = vec3(1.0) - color;
	passTextureCoord = textureCoord;
}
